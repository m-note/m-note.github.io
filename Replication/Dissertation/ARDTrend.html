データの整理

gwf_noautocracyとgwf_regimetypeをまとめる

data$regime <- NA
for (i in 1: nrow(data)){
  if (is.na(data[i, "gwf_regimetype"])){
    if (is.na(data[i, "gwf_nonautocracy"])){
      
    }else(data[i, "regime"] = as.character(data[i, "gwf_nonautocracy"]))
  }else (data[i, "regime"] = as.character(data[i, "gwf_regimetype"]))
}

年ごと、体制ごとにまとめる

体制の分類が細かすぎるので、まとめられるものはまとめる

for (i in 1:nrow(data)){
  if(data[i,"regime"]=="warlord") {data[i,"regime"] <- "warlord/foreign-occupied"}
  if(data[i,"regime"]=="foreign-occupied") {data[i,"regime"] <- "warlord/foreign-occupied"}
  if(data[i,"regime"]=="indirect military") {data[i,"regime"] <- "military"}
  if(data[i,"regime"]=="military-personal") {data[i,"regime"] <- "military"}
  if(data[i,"regime"]=="party-personal") {data[i,"regime"] <- "party"}
  if(data[i,"regime"]=="party-military") {data[i,"regime"] <- "party"}
  if(data[i,"regime"]=="party-military-personal") {data[i,"regime"] <- "party"}
  if(data[i,"regime"]=="not-independent") {data[i,"regime"] <- NA}
}

まとめる

data2 <- data %>%
  dplyr::group_by(year, regime) %>%
  dplyr::summarise(n=n())

割合を求めるために、ある年にある国の合計を求める

year_total <- data %>%
  dplyr::group_by(year) %>%
  dplyr::summarise(total=n())

割合を求める

data2$total <- NA
for(i in 1:nrow(data2)){
  year = as.integer(data2[i, "year"])
  line_num = year - 1960
  data2[i, "total"] <- as.integer(year_total[line_num, "total"])
}

data2$proportion <- (data2$n/data2$total)*100

NAの行はなくす

data3 <- na.omit(data2)

ggplot2でグラフを作成

g <- ggplot(data3) +
  theme_bw() +
  geom_line(mapping=aes(x=year, y=proportion, group=regime, color=regime, ymax=100), position="stack")  +
